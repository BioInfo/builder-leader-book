% Chapter heading styling (native book class, restyled)
\makeatletter
\renewcommand{\@makechapterhead}[1]{%
  \vspace*{60\p@}%
  {\parindent \z@ \raggedright \normalfont
    \ifnum \c@secnumdepth >\m@ne
      \sffamily\fontsize{11pt}{13pt}\selectfont
      \MakeUppercase{\@chapapp}\space\thechapter
      \par\nobreak
      \vskip 10\p@
    \fi
    \rule{\textwidth}{0.4pt}\par
    \vskip 14\p@
    \sffamily\fontsize{22pt}{26pt}\selectfont\itshape #1\par\nobreak

    \vskip 48\p@
  }}

% Drop cap + small caps opening: wrap first paragraph manually with \dropcap{T}{HREE WORDS}
% Usage in markdown: insert a raw latex block at the start of a chapter's first paragraph if desired.
\newcommand{\dropcap}[2]{%
  \noindent%
  {\fontsize{42pt}{42pt}\selectfont\raisebox{-0.35\baselineskip}[0pt][0pt]{#1}}%
  \kern2pt\textsc{#2}\space%
}

% Pull quote environment — triggered by pandoc fenced div ::: pullquote
\newenvironment{pullquote}{%
  \vspace{1.2em}%
  \par\noindent\hspace*{0.05\textwidth}%
  \begin{minipage}{0.9\textwidth}%
    \centering%
    \rule{0.3\textwidth}{0.4pt}\par\vspace{0.8em}%
    \itshape\fontsize{13pt}{18pt}\selectfont%
}{%
    \par\vspace{0.8em}%
    \rule{0.3\textwidth}{0.4pt}%
  \end{minipage}%
  \par\vspace{1.2em}%
}
